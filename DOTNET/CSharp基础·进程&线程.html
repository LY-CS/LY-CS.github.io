# CSharp基础·进程&线程

命名空间：System.Diagnostics;

#新建一个线程对象，并且获取电脑目前所有运行的线程
Process[] pro =  Process.GetProcesses();
foreach(var item in pro)
    {
        Console.WriteLine(item.ProcessName);
    }
Console.ReadKey();

#使用静态方法，调用应用程序
#静态方法的调用是，类名.静态方法名，无需实例化对象
Process.Start("notepad");//打开记事本
Process.Start("mspaint");//打开画图工具
Process.Start("iexplore", "http://www.baidu.com");
Console.ReadKey();

//打开一个文件
//封装我们要打开的文件，但是不打开文件
ProcessStartInfo psi = new ProcessStartInfo(@"C:\Users\tim\Desktop\毕设.txt");
//创建进程对象
Process pro = new Process();
//告诉进程要打开文件的对象
pro.StartInfo = psi;
//打开文件
pro.Start();
Console.ReadKey();
